為處理西洋字母大小寫

方式一

將範圍內所有文字都變更為大寫字母的巨集

Sub Uppercase()

' Loop to cycle through each cell in the specified range.

For Each x In Range("A1:A5")

' Change the text in the range to uppercase letters.

x.Value = UCase(x.value)

Next

End Sub

方式二

將範圍內所有文字都變更為小寫字母的巨集

Sub Lowercase()

' Loop to cycle through each cell in the specified range.

For Each x In Range("B1:B5")

x.Value = LCase(x.Value)

Next

End Sub

Next

End Sub
